\begin{align*}
& [k]: \ 2^k T(n/2^k) \\
& \text{At some point: } \ T(n/2^k) = 1, \ \text{i.e.} \ n/2^k = 1. \\
& n/2^k = 1 \equiv n/2^{\log_2 n} \ \therefore k = \log_2 n \\
& \text{Substitute } \ \log_2 n \ \text{for } \ k \ \text{in} \ [k]: \ {2^{\log_2 n}} \ \times (1) \\
& 2^{\log_2 n} \Leftrightarrow \log_2 2^n = n \times \log_2 2 = n \times 1 \\
& T(n) \ \text{is } \ \Theta(n)
